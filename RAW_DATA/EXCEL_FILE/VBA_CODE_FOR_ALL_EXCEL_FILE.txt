Sub CombineAllFilesIntoSheets()
    Dim FolderPath As String
    Dim Filename As String
    Dim wbSource As Workbook
    Dim wbTarget As Workbook
    Dim SheetName As String
    
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    
    Set wbTarget = ThisWorkbook
    
    FolderPath = "D:\Sobuj\E_COMMERCE_PROJECT\DATT_EXCEL_FILE\"
    
    Filename = Dir(FolderPath & "*.xl*")
    If Filename = "" Then Filename = Dir(FolderPath & "*.csv")
    
    Do While Filename <> ""
        If Filename <> wbTarget.Name Then
            Set wbSource = Workbooks.Open(FolderPath & Filename)
            
            If InStr(Filename, ".") > 0 Then
                SheetName = Left(Filename, InStr(Filename, ".") - 1)
            Else
                SheetName = Filename
            End If
            
            SheetName = Left(SheetName, 30)
            
            wbSource.Sheets(1).Copy After:=wbTarget.Sheets(wbTarget.Sheets.Count)
            wbTarget.Sheets(wbTarget.Sheets.Count).Name = SheetName
            
            wbSource.Close SaveChanges:=False
        End If
        Filename = Dir()
    Loop
    
    If wbTarget.Sheets(1).Name = "Sheet1" And wbTarget.Sheets.Count > 1 Then
        wbTarget.Sheets(1).Delete
    End If
    
    Application.ScreenUpdating = True
    Application.DisplayAlerts = True
End Sub
